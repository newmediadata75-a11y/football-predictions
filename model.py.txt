import pandas as pd
from sklearn.linear_model import LogisticRegression

# 1. قراءة البيانات من fichier CSV
df = pd.read_csv("matches.csv")

# 2. تجهيز البيانات (الأهداف فقط كبداية)
X = df[["home_goals", "away_goals"]]
y = df["result"].map({"HomeWin": 1, "Draw": 0, "AwayWin": -1})

# 3. تدريب النموذج
model = LogisticRegression()
model.fit(X, y)

# 4. تجربة التوقع على مباراة جديدة
# مثال: فريق المضيف سجل 2 أهداف والضيف 0
prediction = model.predict([[2, 0]])

# 5. عرض النتيجة بشكل أوضح
label_map = {1: "فوز المضيف", 0: "تعادل", -1: "فوز الضيف"}
print("توقع النموذج:", label_map[int(prediction[0])])
